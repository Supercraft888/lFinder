function loginCircle(xx, yy, r, aD) {
  var aR = (aD-90) * Math.PI / 180.0;
  return {
    x: xx + (r * Math.cos(aR)),
    y: yy + (r * Math.sin(aR))
  };
}
function loginDrawCircle(x, y, r, sa, ea){
    var s = loginCircle(x, y, r, ea);
    var e = loginCircle(x, y, r, sa);
    var f = ea - sa <= 180 ? "0" : "1";
    return ["M", s.x, s.y, "A", r, r, 0, f, 0, e.x, e.y].join(" ");
}
